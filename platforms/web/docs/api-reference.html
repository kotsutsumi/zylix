Components

Zylix components are functions that return virtual DOM nodes.

h(type, props, ...children)

Creates a virtual DOM element.

function h(
  type: string | Function,
  props: object | null,
  ...children: Array<VNode | string>
): VNode
Parameter Type Description
type string | Function Element tag name or component function
props object | null Element properties and attributes
children VNode[] Child elements or text content
Example
// Create a simple element
const button = h('button', { onClick: handleClick }, 'Click me');

// Create a component
function Card({ title, children }) {
  return h('div', { className: 'card' },
    h('h2', null, title),
    h('div', { className: 'card-body' }, ...children)
  );
}

// Use the component
const myCard = h(Card, { title: 'Hello' },
  h('p', null, 'Card content')
);

render(vnode, container)

Renders a virtual DOM tree into a DOM container.

function render(
  vnode: VNode,
  container: HTMLElement
): void
Example
function App() {
  return h('div', null,
    h('h1', null, 'Hello, Zylix!')
  );
}

render(h(App), document.getElementById('app'));

Hooks

Hooks enable state and lifecycle features in functional components.

useState(initialValue)

Adds local state to a component.

function useState<T>(
  initialValue: T | () => T
): [T, (value: T | (prev: T) => T) => void]
Returns

A tuple containing the current state value and a setter function.

Example
function Counter() {
  const [count, setCount] = useState(0);

  return h('div', null,
    h('p', null, `Count: ${count}`),
    h('button', { onClick: () => setCount(c => c + 1) }, '+'),
    h('button', { onClick: () => setCount(c => c - 1) }, '-')
  );
}

useEffect(effect, deps)

Runs side effects after render.

function useEffect(
  effect: () => void | (() => void),
  deps?: any[]
): void
Parameter Type Description
effect Function Effect function, optionally returns cleanup
deps optional any[] Dependency array (effect re-runs when deps change)
Example
function Timer() {
  const [seconds, setSeconds] = useState(0);

  useEffect(() => {
    const interval = setInterval(() => {
      setSeconds(s => s + 1);
    }, 1000);

    // Cleanup function
    return () => clearInterval(interval);
  }, []); // Empty deps = run once on mount

  return h('p', null, `Elapsed: ${seconds}s`);
}

useMemo(factory, deps)

Memoizes expensive computations.

function useMemo<T>(
  factory: () => T,
  deps: any[]
): T
Example
function ExpensiveList({ items, filter }) {
  const filteredItems = useMemo(() => {
    return items.filter(item => item.name.includes(filter));
  }, [items, filter]);

  return h('ul', null,
    ...filteredItems.map(item => h('li', { key: item.id }, item.name))
  );
}

useCallback(callback, deps)

Memoizes callback functions.

function useCallback<T extends Function>(
  callback: T,
  deps: any[]
): T

useRef(initialValue)

Creates a mutable ref object.

function useRef<T>(
  initialValue: T
): { current: T }
Example
function TextInput() {
  const inputRef = useRef(null);

  const focusInput = () => {
    inputRef.current?.focus();
  };

  return h('div', null,
    h('input', { ref: inputRef, type: 'text' }),
    h('button', { onClick: focusInput }, 'Focus')
  );
}

Store

Global state management with actions, selectors, and middleware support.

createStore(config)

Creates a global state store.

function createStore<T>(
  config: {
    initialState: T,
    actions: Record<string, (state: T, payload?: any) => T>,
    selectors?: Record<string, (state: T) => any>,
    middleware?: Middleware[]
  }
): Store<T>
Example
const store = createStore({
  initialState: {
    count: 0,
    todos: []
  },

  actions: {
    increment: (state) => ({ ...state, count: state.count + 1 }),
    decrement: (state) => ({ ...state, count: state.count - 1 }),
    addTodo: (state, text) => ({
      ...state,
      todos: [...state.todos, { id: Date.now(), text, done: false }]
    })
  },

  selectors: {
    count: (state) => state.count,
    todos: (state) => state.todos,
    activeTodos: (state) => state.todos.filter(t => !t.done)
  }
});

store.dispatch(action, payload)

Dispatches an action to update state.

store.dispatch(
  action: string,
  payload?: any
): void

store.select(selector)

Selects a slice of state using a selector.

store.select(
  selector: string | (state: T) => any
): any

store.subscribe(listener)

Subscribes to state changes.

store.subscribe(
  listener: (state: T) => void
): () => void
Returns

Unsubscribe function.

useStore(store, selector)

Hook to use store state in components.

function useStore<T, R>(
  store: Store<T>,
  selector: string | (state: T) => R
): R
Example
function Counter() {
  const count = useStore(store, 'count');

  return h('div', null,
    h('span', null, `Count: ${count}`),
    h('button', { onClick: () => store.dispatch('increment') }, '+')
  );
}

HTTP Client

Built-in HTTP client with interceptors and request/response handling.

http.get(url, options)

Performs a GET request.

async function http.get<T>(
  url: string,
  options?: RequestOptions
): Promise<T>

http.post(url, data, options)

Performs a POST request.

async function http.post<T>(
  url: string,
  data: any,
  options?: RequestOptions
): Promise<T>

http.put / http.patch / http.delete

Similar to post, for PUT, PATCH, and DELETE methods.

http.addInterceptor(interceptor)

Adds a request/response interceptor.

http.addInterceptor(
  interceptor: {
    request?: (config) => config,
    response?: (response) => response,
    error?: (error) => Promise
  }
): void
Example
// Add auth header to all requests
http.addInterceptor({
  request: (config) => {
    config.headers = {
      ...config.headers,
      'Authorization': `Bearer ${getToken()}`
    };
    return config;
  }
});

// Fetch data
const users = await http.get('/api/users');
const newUser = await http.post('/api/users', { name: 'John' });

Router

Client-side routing with dynamic routes and navigation guards.

createRouter(routes)

Creates a router instance.

function createRouter(
  routes: Route[]
): Router
Example
const router = createRouter([
  { path: '/', component: Home },
  { path: '/users', component: UserList },
  { path: '/users/:id', component: UserDetail },
  { path: '/admin/*', component: AdminLayout, guard: requireAuth },
  { path: '*', component: NotFound }
]);

// Navigate programmatically
router.push('/users/123');
router.replace('/login');
router.back();

useRouter()

Hook to access router in components.

function useRouter(): {
  route: Route,
  params: Record<string, string>,
  query: Record<string, string>,
  push: (path: string) => void,
  replace: (path: string) => void,
  back: () => void
}

Link

Navigation component.

Example
function Nav() {
  return h('nav', null,
    h(Link, { to: '/' }, 'Home'),
    h(Link, { to: '/users' }, 'Users'),
    h(Link, { to: '/about', className: 'nav-link' }, 'About')
  );
}

Testing Library

Comprehensive testing utilities for Zylix components.

render(component)

Renders a component for testing.

function render(
  component: VNode
): {
  container: HTMLElement,
  debug: () => void,
  rerender: (newComponent: VNode) => void,
  unmount: () => void
}
Example
import { render, screen } from 'zylix/testing';

test('renders greeting', () => {
  render(h(Greeting, { name: 'World' }));
  expect(screen.getByText('Hello, World!')).toBeInTheDocument();
});

screen

Query methods for finding elements.

Method Description
getByText(text) Find by text content (throws if not found)
getByTestId(id) Find by data-testid attribute
getByRole(role) Find by ARIA role
getByPlaceholder(text) Find input by placeholder
getByLabelText(text) Find input by associated label
queryBy*() Same as getBy but returns null if not found
getAllBy*() Returns array of all matching elements
findBy*() Async version, waits for element

fireEvent

Fires DOM events on elements.

fireEvent.click(element: HTMLElement): void
fireEvent.change(element, { target: { value } }): void
fireEvent.submit(element): void
fireEvent.keyDown(element, { key, code }): void

userEvent

Simulates real user interactions.

await userEvent.click(element): Promise
await userEvent.type(element, text): Promise
await userEvent.clear(element): Promise
await userEvent.selectOptions(select, values): Promise
Example
test('user can submit form', async () => {
  render(h(LoginForm));

  await userEvent.type(screen.getByPlaceholder('Email'), 'test@example.com');
  await userEvent.type(screen.getByPlaceholder('Password'), 'secret123');
  await userEvent.click(screen.getByRole('button', { name: 'Login' }));

  expect(await screen.findByText('Welcome!')).toBeInTheDocument();
});

waitFor(callback)

Waits for async conditions.

async function waitFor(
  callback: () => void,
  options?: { timeout?: number, interval?: number }
): Promise<void>
Example
await waitFor(() => {
  expect(screen.getByText('Loaded!')).toBeInTheDocument();
}, { timeout: 3000 });

createMockStore(config)

Creates a mock store for testing.

function createMockStore(
  config: StoreConfig
): MockStore
Returns

Mock store with action tracking: getActions(), getLastAction(), clearActions()

createMockServer()

Creates a mock HTTP server for testing.

function createMockServer(): {
  mock: (method, url, response, options?) => void,
  getRequests: () => Request[],
  reset: () => void,
  restore: () => void
}
Example
const server = createMockServer();

// Mock API responses
server.mock('GET', '/api/users', [
  { id: 1, name: 'Alice' },
  { id: 2, name: 'Bob' }
]);

server.mock('POST', '/api/users', { id: 3, name: 'Charlie' });

// With delay
server.mock('GET', '/api/slow', { data: 'result' }, { delay: 1000 });

// Verify requests
const requests = server.getRequests();
expect(requests[0].url).toBe('/api/users');

// Cleanup
server.restore();

expect() Matchers

Assertion matchers for testing.

Matcher Description
toBe(value) Strict equality (===)
toEqual(value) Deep equality
toBeTruthy() Value is truthy
toBeFalsy() Value is falsy
toBeNull() Value is null
toContain(item) Array/string contains item
toHaveLength(n) Array/string has length
toBeGreaterThan(n) Number comparison
toBeInTheDocument() Element is in DOM
toHaveTextContent(text) Element contains text
toHaveAttribute(name, value?) Element has attribute
toHaveClass(className) Element has CSS class
toBeVisible() Element is visible
toBeDisabled() Element is disabled
toHaveValue(value) Input has value
toThrow(message?) Function throws error
.not Negates any matcher
Note

All testing utilities are tree-shakeable. Import only what you need for optimal bundle size.
