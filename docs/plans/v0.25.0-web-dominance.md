  state: {
    todos: [],
    filter: 'all'
  },

  actions: {
    addTodo(state, text) {
      return { ...state, todos: [...state.todos, { id: Date.now(), text, done: false }] };
    },
    toggleTodo(state, id) {
      return {
        ...state,
